    Entradas: Numero (N)
    Proceso: 
        5.1 ingresar el Valor seleccionado (N) 
        5.2 if (N menor que 1)
            5.2.1 imprimir "no es primo"
        5.3 if (N = 2)
            5.3.1 imprimir "es primo"
        5.4 if (N % 2 (residuo)) = 0
            5.4.1 imprimir "no es primo"
        5.5 else
            5.5.1 iniciar i = 3
            5.5.2 repetir mientras i <= raiz de N
                5.5.2.1 if (N % i = 0)
                    5.5.2.1.1 imprimir "no es primo"
                    5.5.2.1.2 salir del ciclo
                5.5.2.2 incrementar i en 2
            5.5.3 si no se encontró divisor
                5.5.3.1 imprimir "es primo"
    Salida: 
        "es primo o no"
